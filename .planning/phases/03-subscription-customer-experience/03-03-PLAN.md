**handleAction function:**
```typescript
async function handleAction(subscriptionId: string, action: 'pause' | 'resume' | 'skip' | 'cancel') {
  setLoadingAction({ id: subscriptionId, action });
  try {
    const res = await fetch(`/api/subscriptions/${subscriptionId}/${action}`, {
      method: 'POST',
    });
    if (!res.ok) {
      const data = await res.json();
      throw new Error(data.error || `Failed to ${action} subscription`);
    }
    toast.success(actionSuccessMessages[action]);
    router.refresh(); // Refresh server component data
  } catch (error) {
    toast.error(error instanceof Error ? error.message : `Failed to ${action} subscription`);
  } finally {
    setLoadingAction(null);
  }
}
```
